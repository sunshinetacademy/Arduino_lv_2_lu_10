💡 الحل العبقري: الصفوف والأعمدة

الأردوينو يلعب لعبة "المصافحة":

  1. 1 يشغل الصف الأول (يُعطيه طاقة).
  2. 2 يسأل الأعمدة: "هل ضغط أحد على زر في هذا الصف؟"
  3. 3 إذا أجاب العمود الثالث بنعم، يعرف الأردوينو فوراً أن الزر هو "3"!
4 صفوف + 4 أعمدة = 8 أسلاك فقط! 🎉

🧪 الكود 1: قراءة الأزرار وعرضها

#include <Keypad.h>

// 1. تعريف شكل المصفوفة
const byte ROWS = 4;
const byte COLS = 4;

// 2. رسم خريطة الأزرار
char hexaKeys[ROWS][COLS] = {
  {'1', '2', '3', 'A'},
  {'4', '5', '6', 'B'},
  {'7', '8', '9', 'C'},
  {'*', '0', '#', 'D'}
};

// 3. تعريف أسلاك الأردوينو
byte rowPins[ROWS] = {9, 8, 7, 6};
byte colPins[COLS] = {5, 4, 3, 2};

// 4. إنشاء كائن لوحة المفاتيح
Keypad customKeypad = Keypad(
  makeKeymap(hexaKeys), rowPins, colPins, ROWS, COLS
);

void setup() {
  Serial.begin(9600);
}

void loop() {
  char key = customKeypad.getKey();
  
  if (key) { // إذا تم ضغط زر
    Serial.print("Pressed: ");
    Serial.println(key);
  }
}
